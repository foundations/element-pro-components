# Select

封装选择器

# 使用

<template>
  <pro-select
    v-model="select"
    :data="list"
    :config="config"
  />
</template>

<script>
import { ref } from 'vue'

export default {
  setup() {
    const select = ref('')
    const config = ref({ label: 'tag' })
    const list = ref([
      { value: 'Go', tag: 'go', disabled: true },
      { value: 'JavaScript', tag: 'javascript' },
      { value: 'Python', tag: 'python' },
    ])
    return {
      select,
      config,
      list,
    }
  }
}
</script>
